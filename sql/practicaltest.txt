2)
WITH HIGHEST AS (
SELECT DEPARTMENT,
FIRST_NAME ,
LAST_NAME,
SALARY,
DENSE_RANK() OVER(PARTITION BY DEPARTMENT ORDER BY SALARY DESC) AS RowNum
FROM Worker_Details
)
SELECT DEPARTMENT, FIRST_NAME,LAST_NAME, SALARY
FROM HIGHEST
WHERE RowNum = 1;
